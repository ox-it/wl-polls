#!/bin/bash

ALL_KEYS=$(find . -name '*.properties' -a ! -name project.properties -exec sed -e 's/#.*//' -e 's/[[:space:]]*//g' \{\} \;|fgrep '=' | awk -F'=' '{print $1}' |sort|uniq)

for KEY in $ALL_KEYS
do 
  if ! find . \( -name .svn -prune -o -name target -prune -o -type f \) -printf \'%h/%f\'\  | xargs fgrep -l ${KEY} |fgrep -v '.properties' | fgrep -v '.metaprops' >/dev/null
	then
		echo ${KEY}
	fi
done
